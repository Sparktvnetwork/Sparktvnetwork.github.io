// Add tab click event listeners document.querySelectorAll('.tab').forEach(tab => { tab.addEventListener('click', () => { document.querySelectorAll('.tab').forEach(t => t.classList.remove('active')); tab.classList.add('active'); document.querySelectorAll('.tab-content > div').forEach(content => content.style.display = 'none'); document.querySelector(`#${tab.id}-content`).style.display = 'block'; }); });
